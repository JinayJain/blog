using namespace std;

int main()
{

    int T;
    cin >> T;

    for (int t = 0; t < T; t++)
    {
        int n, a, b, c, d;
        cin >> n >> a >> b >> c >> d;

        double avgmin = (double)(c - d) / (double)n;
        double avgmax = (double)(c + d) / (double)n;

        if (avgmin - 1e-8 <= (double)(a + b) && avgmax + 1e-8 >= (double)(a - b))
        {
            printf("Yes\n");
        }
        else
        {
            printf("No\n");
        }
    }

    return 0;
}
